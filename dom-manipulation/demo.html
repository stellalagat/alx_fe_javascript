JSON Export Implementation

function exportQuotes() { const data = { quotes: quotes, exportDate: new Date().toISOString(), totalQuotes: quotes.length }; const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); // Create download link const link = document.createElement('a'); link.href = url; link.download = 'quotes-export.json'; link.click(); }
